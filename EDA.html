Variable statistics function¶

In [48]:
def print_statistics(col):
    print("mean: {0}".format(col.mean()))
    print("sd: {0}".format(col.std()))
    print("max: {0}".format(col.max()))
    print("min: {0}".format(col.min()))
    print("median: {0}".format(col.median()))
    print("mode: {0}".format(col.mode()[0]))
